\begin{align*}
    \alpha & = 10.2642\ \text{s}^2/\text{kg} \\
    \beta & = 0.0629\ \text{s}^2 \\
    r & = 1.00000
\end{align*}

计算:\[ T^2 = \frac{4\pi^2}{k}m_1 + \frac{4\pi^2}{k}m_0 \],
\begin{enumerate}
    \item \textbf{倔强系数 $k$:}
    \[ k = \frac{4\pi^2}{\alpha} = \frac{4\pi^2}{10.2642} = 3.848\ \text{N/m} \]
    
    \item \textbf{弹簧的有效质量 $m_0$:}
    \[ m_0 = \frac{\beta}{\alpha} = \frac{0.0629}{10.2642} = 0.00613\ \text{kg} = 6.13\ \text{g} \]
    
    \item \textbf{相对误差:}
    \begin{enumerate}
        \item 理论值 $m_{0,th} = m_s/3 = 21.18 / 3 = 7.06$ g
        \[ E_r(m_0) = \frac{|m_0 - m_{0,th}|}{m_{0,th}} = \frac{|6.13 - 7.06|}{7.06} \times 100\% = 13.2\% \]
        \item  $\bar{k} = (3.867+3.865+3.866+3.865)/4 = 3.866$ N/m 
        \[ E_r(k) = \frac{|k - \bar{k}|}{\bar{k}} = \frac{|3.848 - 3.866|}{3.866} \times 100\% = 0.47\% \]
    \end{enumerate}
\end{enumerate}
